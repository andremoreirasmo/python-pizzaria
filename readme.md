#Projeto Pizzaria
Projeto pizzaria simples feito em Django.

#Criar env
python3 -m venv env

#Criar usuario
python3 manage.py createsuperuser

#criar app
django-admin startapp <nomedoapp>

#Rodar server
python3 manage.py runserver

#Criar migrattions
python3 manage.py makemigrations

#Rodar migrates
python3 manage.py migrate
